<Playground>
  {() => {
    const App = () => {
      const [type, setType] = React.useState(types[0])
      function handleClick() {
        const filtered = types.filter((_type) => _type !== type)
        setType(filtered[Math.floor(Math.random() * filtered.length)])
      }
      return (
        <div style={styles}>
          <Tag type={type} onClick={handleClick}>
            {type}
          </Tag>
        </div>
      )
    }
    return <App />
  }}
</Playground>
